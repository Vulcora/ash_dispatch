    AshDispatch.Behaviours.CounterBroadcaster behaviour (AshDispatch v0.1.0)

    View Source

    Behaviour for broadcasting real-time counter updates to users.

    This behaviour allows applications to implement their own counter broadcasting logic, from simple Phoenix PubSub to complex multi-tenant WebSocket systems.

    Counter System Overview

    The counter system provides real-time updates for:

    • User-specific counts (cart items, unread notifications)
    • Admin totals (pending orders, open tickets)
    • Mixed user/admin counts (pending tickets visible to both)

    Implementation

    Applications implement this behaviour to bridge AshDispatch with their real-time communication layer (Phoenix Channels, Absinthe subscriptions, etc.).

    Simple Implementation

    defmodule MyApp.CounterBroadcaster do
  @behaviour AshDispatch.Behaviours.CounterBroadcaster

  def broadcast_user_counter(user_id, counter_name, count, metadata) do
    MyAppWeb.Endpoint.broadcast(
      "user:#{user_id}",
      "counter:update",
      %{counter: counter_name, value: count, metadata: metadata}
    )
    :ok
  end

  def broadcast_admin_counter(admin_id, counter_name, count, metadata) do
    MyAppWeb.Endpoint.broadcast(
      "user:#{admin_id}",
      "admin:counter:update",
      %{counter: counter_name, value: count, metadata: metadata}
    )
    :ok
  end

  def get_admin_ids(opts) do
    query = MyApp.Accounts.User |> Ash.Query.filter(admin == true)

    query = if exclude = opts[:exclude_user_id] do
      Ash.Query.filter(query, id != ^exclude)
    else
      query
    end

    case Ash.read(query, authorize?: false) do
      {:ok, admins} -> {:ok, Enum.map(admins, & &1.id)}
      error -> error
    end
  end
end

    Advanced Implementation

    For multi-tenant or complex routing:

    defmodule MyApp.AdvancedCounterBroadcaster do
  @behaviour AshDispatch.Behaviours.CounterBroadcaster

  def broadcast_user_counter(user_id, counter_name, count, metadata) do
    # Route based on tenant, add encryption, batch updates, etc.
    tenant = metadata[:tenant_id]
    MyApp.TenantBroadcaster.broadcast(tenant, user_id, counter_name, count)
  end

  # ... other callbacks
end

    Configuration

    Configure the counter broadcaster in your app's config:

    config :ash_dispatch,
  counter_broadcaster: MyApp.CounterBroadcaster

    Metadata

    The metadata map may contain:

    • :invalidate_queries - List of query keys to invalidate in frontend cache
    • :tenant_id - For multi-tenant applications
    • Custom application-specific metadata

    Callbacks

    broadcast_admin_counter(admin_id, counter_name, count, metadata)

    @callback broadcast_admin_counter(
  admin_id :: String.t(),
  counter_name :: atom(),
  count :: non_neg_integer(),
  metadata :: map()
) :: :ok | {:error, term()}

    Broadcast an admin counter update to a specific admin.

    Admin counters are typically prefixed with admin_ (e.g., :admin_pending_orders) and represent totals across all users.

    Parameters

    • admin_id - Admin user UUID to broadcast to
    • counter_name - Admin counter identifier (e.g., :admin_pending_orders)
    • count - Current total count
    • metadata - Optional metadata map

    Returns

    :ok or {:error, reason}

    broadcast_user_counter(user_id, counter_name, count, metadata)

    @callback broadcast_user_counter(
  user_id :: String.t(),
  counter_name :: atom(),
  count :: non_neg_integer(),
  metadata :: map()
) :: :ok | {:error, term()}

    Broadcast a counter update to a specific user.

    Parameters

    • user_id - User UUID to broadcast to
    • counter_name - Counter identifier (e.g., :cart_items, :pending_orders)
    • count - Current count value
    • metadata - Optional metadata map (query invalidation, tenant info, etc.)

    Returns

    :ok or {:error, reason}

    get_admin_ids(opts)

    @callback get_admin_ids(opts :: keyword()) :: {:ok, [String.t()]} | {:error, term()}

    Get list of admin user IDs.

    Used for broadcasting admin counters to all administrators.

    Options

    • :exclude_user_id - User ID to exclude from results (prevents duplicate broadcast when user is admin)

    Returns

    {:ok, [user_id]} or {:error, reason}
